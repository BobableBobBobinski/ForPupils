#include <iostream>
using namespace std;
void fill(double* m, int n)
{
	for (int i = 0; i < n; i++)
	{
		m[i] = rand() % 100 / 3.14;
	}
}
double*  delete_by_index(double* mas, int n, int index)
{

	double* buf = new double[n - 1];
	
	for (int i = 0; i < index; i++)
	{
		buf[i] = mas[i];
	}
	for (int i = index + 1; i < n; i++)
	{
		buf[i - 1] = mas[i];
	}
	delete[] mas;
	
	return buf;

}
