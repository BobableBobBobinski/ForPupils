#include <iostream>
#include <vector>
#include <fstream>
#include <iomanip>

using namespace std;
float Heron(int a, int b, int c)
{
	float p = (a + b + c) / 2;
	return sqrt(p*(p - a)*(p - b)*(p - c));
}

int main()
{
	setlocale(LC_ALL, "Rus");
	ofstream fout;
	ifstream fin;

	fout.open("output.txt");
	fin.open("input.txt");
	int a;
	int b;
	int c;

	while (!fin.eof())
	{
		fin >> a;
		fin >> b;
		fin >> c;
		if (a + b > c & c + b > a && a + c > b)
		{
			
			fout << a << " " << b << " " << c << " " << Heron(a, b, c) << endl;
		}
		else
		{
			fout << a << " " << b << " " << c << " NO" << endl;
		}

	}










	if ((a + b + c) / 2 > a || (a + b + c) / 2 > b || (a + b + c) / 2 > c)
	{
		fout << "a     b     (a + b + c) / 2     " << endl;
		for (int i = 0; i < 6; i++)
		{
			fin >> a;
			fin >> b;
			fout << left << setfill('#') << a << setw(5) << b << setw(5) << (a + b + c) / 2 << setw(5) << endl;
		}

	}
	else
	{
		fout << "a     b     (a + b + c) / 2     " << endl;
		for (int i = 0; i < 6; i++)
		{
			fin >> a;
			fin >> b;
			fout << left << setfill('#') << a << setw(5) << b << setw(5) << "NO >=[" << setw(5) << endl;
		}
	}
	fout.close();
	fin.close();
	system("pause");
}
